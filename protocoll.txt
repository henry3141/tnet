api endpoints:

/make_route - POST
    -ARGS:
        -start_pos:(number,number,number)
        -end_pos:(number,number,number)
        -start_time:(number)
    -RETURNS:
        -route_id:number
        -route_end_time:number

/completed_step - POST 
    -ARGS:
        -route_id:number
    -RETURNS:
        -OK 

/next_step - GET
    -ARGS:
        -route_id:number
    -RETURNS:
        -step:(number,number,number) //nur eins kann was anderes als 0 sein
        -step_time:number

/no_go - POST 
    -ARGS:
        -pos:(number,number,number)
    -RETURNS:
        -OK

/ok_go - POST 
    -ARGS:
        -pos:(number,number,number)
    -RETURNS:
        -OK
